function! CerrarLlave(lista)
  let linea = getline('.')
  let pos = col('.')
  if pos < col('$')
    return "{"
  endif
  for cad in a:lista
    if (stridx(linea, cad, pos - len(cad) - 1) != -1)
      return " {\<CR>}\<C-o>==\<C-o>O"
    endif
  endfor
  return "{}\<left>"
endfunction
